    **Part B: Create CandidateFilterStage**

    Create `stream/stages/generation/candidate_filter_stage.py`:

    ```python
    import logging

    from stream.datatypes import LayerDim
    from stream.opt.search_space import SearchSpace, TileSizeOption
    from stream.opt.tile_size_utils import is_divisible_candidate, passes_single_tensor_memory_check
    from stream.stages.context import StageContext
    from stream.stages.stage import Stage, StageCallable

    logger = logging.getLogger(__name__)


    class CandidateFilterStage(Stage):
        """Filter candidate tile sizes for divisibility and single-tensor memory feasibility.

        Reads tile_options_raw (dict[str, list[int]]) from ctx, resolves string dim names
        to unique LayerDim groups, filters candidates, builds a SearchSpace, and sets it
        into ctx for downstream CO stages.

        Per D-06: runs before the CO stage. Per D-03: divisibility is mandatory.
        Per D-04: single-tensor memory check. Per D-05: no full allocation feasibility.
        Per D-10: keyed by unique dimension group, not individual LayerDim names.
        """

        REQUIRED_FIELDS = ("workload", "mapping", "accelerator", "tile_options_raw")

        def __init__(self, list_of_callables: list[StageCallable], ctx: StageContext):
            super().__init__(list_of_callables, ctx)
            self.workload = self.ctx.get("workload")
            self.mapping = self.ctx.get("mapping")
            self.accelerator = self.ctx.get("accelerator")
            self.tile_options_raw: dict[str, list[int]] = self.ctx.get("tile_options_raw")

        def run(self):
            search_space = self._build_search_space()
            self.ctx.set(search_space=search_space)
            sub_stage = self.list_of_callables[0](self.list_of_callables[1:], self.ctx)
            yield from sub_stage.run()

        def _build_search_space(self) -> SearchSpace:
            search_space = SearchSpace()

            # Resolve string dim names -> unique LayerDim groups (per D-10, D-11)
            # tile_options_raw keys are "NodeName.D0" strings
            # FusedGroup.intra_core_tiling has (LayerDim, tile) tuples from factory
            # The factory already resolved "NodeName.D0" -> unique LayerDim
            # We need to do the same resolution here for tile_options_raw keys
            dim_name_to_unique: dict[str, LayerDim] = {}
            fused_group = self.mapping.fused_groups[0]  # single fused group (asserted in determine_fusion_splits)
            for dim, _tile in fused_group.intra_core_tiling:
                # dim is already a unique LayerDim (resolved by MappingFactory._convert_intra_core_tiling_entry)
                # We need to map back from raw dim string to this LayerDim
                pass

            # Better approach: iterate over tile_options_raw keys and resolve them
            # the same way MappingFactory._convert_intra_core_tiling_entry does
            for raw_dim_str, candidates in self.tile_options_raw.items():
                # Resolve raw_dim_str (e.g. "Gemm_Left.D1") to unique LayerDim
                node_name, dim_name = raw_dim_str.split(".")
                dim_idx = int(dim_name[1:])  # "D1" -> 1
                node = self.workload.get_node_by_name(node_name)
                unique_dim = self.workload.get_dims(node)[dim_idx]
                workload_size = self.workload.get_dimension_size(unique_dim)

                # Skip if we already processed this unique dim (multiple LayerDims may map to same unique dim)
                if unique_dim in [d for d in search_space.dims()]:
                    continue

                for candidate in candidates:
                    # D-03: Divisibility check (accounts for spatial unrolling per Pitfall 2)
                    if not is_divisible_candidate(self.workload, self.mapping, unique_dim, candidate):
                        logger.info(
                            "Candidate tile %d for dim %s rejected: not divisible "
                            "(workload_size=%d)",
                            candidate, unique_dim, workload_size,
                        )
                        continue

                    option = TileSizeOption(dim=unique_dim, tile=candidate, workload_size=workload_size)
                    search_space.add(unique_dim, option)
                    logger.info(
                        "Candidate tile %d for dim %s accepted",
                        candidate, unique_dim,
                    )

            # D-04: Single-tensor memory check per compute node
            # Get the smallest compute core's memory capacity as conservative bound (per Anti-Pattern note in Research)
            compute_cores = [
                core for core in self.accelerator.cores
                if hasattr(core, 'get_memory_capacity')
            ]
            if compute_cores:
                for node in self.workload.get_computation_nodes():
                    for core in compute_cores:
                        if not passes_single_tensor_memory_check(self.workload, self.mapping, node, core):
                            logger.warning(
                                "Node %s has a tensor exceeding memory capacity of core %s",
                                node.name, core,
                            )
                            # Note: This is a per-node check, not per-candidate.
                            # The memory check verifies post-inter-core-split tensor sizes
                            # which are independent of the intra-core tile candidate.
                            # If a node fails, it means the mapping itself is infeasible.
                            break

            # Pitfall 5: Assert non-empty SearchSpace
            for dim in search_space.dims():
                if not search_space.get(dim):
                    raise ValueError(
                        f"All candidate tiles filtered out for dimension {dim}. "
                        f"No valid tile sizes remain. Check divisibility and memory constraints."
                    )

            if search_space.is_empty():
                raise ValueError(
                    "SearchSpace is empty â€” no valid tile candidates for any dimension. "
                    "Ensure tile_options in the mapping YAML contains at least one valid divisor "
                    "per workload dimension."
                )

            return search_space
    ```
