    ```python
    def get_selected_tiles(self) -> dict[LayerDim, int]:
        """Return {dim: tile_size} for each dimension in the search space, post-solve."""
        if not self.w:
            return {}
        result: dict[LayerDim, int] = {}
        for dim in self.search_space.dims():
            options = self.search_space.get(dim)
            for k, opt in enumerate(options):
                if self.w[(dim, k)].X > self.VAR_THRESHOLD:
                    result[dim] = opt.tile
                    break
        return result
    ```

    ```python
    def substitute_loop_sizes_with_selected_tiles(self) -> dict[LayerDim, int]:
        """Build tiled_sizes dict from CO-solved selected tiles."""
        unique_dims, _ = self.workload.unique_dimensions()
        _, unique_spatial_unrollings = collect_spatial_unrollings(self.workload, self.mapping)
        unique_unrollings_dict = dict(unique_spatial_unrollings)
        result = {}
        for dim, tile_size in self.selected_tiles.items():
            spatial = unique_unrollings_dict.get(dim, 1)
            # The tiled workload dimension = tile_size * spatial_unrolling
            # (same logic as determine_fusion_splits but in reverse: tile_size is now known)
            result[dim] = tile_size * spatial
        # Non-tiled dimensions keep their original size
        for dim in set(unique_dims) - set(self.selected_tiles.keys()):
            result[dim] = self.workload.get_dimension_size(dim)
        return result
    ```

    **Approach: Add a lightweight inline stage or compute fusion_splits inside the stage pipeline.**

    The cleanest approach per the research: create a small `FusionSplitsComputationStage` or compute it as a pre-step. Since `determine_fusion_splits()` needs `workload` and `mapping` which are only set by `MappingParserStage`, we need this to happen after MappingParserStage.

    Create a minimal stage class (can be in `api.py` or a new file). Recommended: add it inline in `api.py` or as a small class:

    ```python
    class _FusionSplitsStage(Stage):
        """Compute fusion_splits from untiled workload and mapping.

        This stage replaces the side-effect of TilingGenerationStage that previously
        ran before the CO. It calls determine_fusion_splits() and sets the result
        on ctx so ConstraintOptimizationAllocationStage can read it.
        """
        REQUIRED_FIELDS = ("workload", "mapping")

        def __init__(self, list_of_callables, ctx):
            super().__init__(list_of_callables, ctx)
            self.workload = self.ctx.get("workload")
            self.mapping = self.ctx.get("mapping")

        def run(self):
            from stream.workload.utils import determine_fusion_splits
            fusion_splits = determine_fusion_splits(self.workload, self.mapping)
            self.ctx.set(fusion_splits=fusion_splits)
            sub_stage = self.list_of_callables[0](self.list_of_callables[1:], self.ctx)
            yield from sub_stage.run()
    ```

    ```python
        ConstraintOptimizationAllocationStage,
        TilingGenerationStage,         # Post-solve: applies selected tiles
    ```
